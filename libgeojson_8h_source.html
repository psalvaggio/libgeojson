32inline const char* TypeName(Type type) {
33 static constexpr auto kPointName = "Point";
34 static constexpr auto kMultiPointName = "MultiPoint";

404inline bool IsCcw(const nlohmann::json& coords) {
405 // Sum (x2 - x1)(y2 + y1), that will be > 0, if the points are CW
406 double cwEdgeSum = 0;
407 for (size_t i = 0; i < coords.size(); i++) {
408 const auto& pt1 = coords[i];
409 const auto& pt2 = coords[(i + 1) % coords.size()];
410
411 cwEdgeSum +=
412 (pt2[0].template get<double>() - pt1[0].template get<double>()) *
413 (pt2[1].template get<double>() + pt1[1].template get<double>());
414 }
415
416 return cwEdgeSum < 0;
417}

429template <typename GetPoint>
430nlohmann::json LinearRingCoordinates(size_t numPoints, bool ccw,
431 GetPoint&& getPoint) {
432 // We must be at least a triangle
433 if (numPoints < 3) {
434 throw std::domain_error("Linear rings must have at least 3 points");
435 }
436
437 // A linear ring is essentailly a line string
438 auto coords =
439 LineStringCoordinates(numPoints, std::forward<GetPoint>(getPoint));
440
441 // Reverse if need it
442 bool isCcw = IsCcw(coords);
443 bool reverse = (isCcw && !ccw) || (!isCcw && ccw);
444 if (reverse) {
445 for (size_t i = 0; i < coords.size() / 2; i++) {
446 std::swap(coords[i], coords[coords.size() - i - 1]);
447 }
448 }
449
450 // Close the ring
451 coords.push_back(coords[0]);
452
453 return coords;
454}
